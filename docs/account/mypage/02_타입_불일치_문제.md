```typescript
// API 응답 타입 (백엔드)
interface ApiAccountResponse {
    id: number;
    email: string;
    userName: string;
    phoneNumber: string;
    accountType: AccountType;
    approvalStatus: ApprovalStatus;  // 백엔드 필드명
}

// 프론트엔드 타입으로 변환
function convertToAccount(apiResponse: ApiAccountResponse): Account {
    return {
        ...apiResponse,
        accountApproved: apiResponse.approvalStatus,  // 필드명 변환
    };
}
```

**장점:**
- ✅ 기존 코드 수정 불필요
